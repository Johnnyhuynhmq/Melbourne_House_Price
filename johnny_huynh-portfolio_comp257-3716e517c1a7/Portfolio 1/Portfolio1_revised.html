# iterate all trkpt elements - the children of trkseg
data = []
for trkpt in trkseg:
    variables = []
    #get some properties from the attributesd
    lon= trkpt.attrib['lon']
    lat = trkpt.attrib['lat']
    # get values from the child elements
    ele = trkpt.find('gpx:ele', ns).text
    time = trkpt.find('gpx:time', ns).text
    # now dive into the extensions
    ext = trkpt.find('gpx:extensions', ns)
    power = ext.find('gpx:power', ns).text 
    tpext = ext.find('gpxtpx:TrackPointExtension', ns)
    temp = tpext.find('gpxtpx:atemp', ns).text
    hr = tpext.find('gpxtpx:hr', ns).text
    cad = tpext.find('gpxtpx:cad', ns).text
    variables = [lon, lat, ele, time, power, temp, hr, cad]
    data.append(variables)
    
# read data into a dataframe and name the variables
columns = ['Longtitude', 'Latitude', 'Elevation', 'Time', 'Power', 'Temperature', 'HeartRate', 'Cadence']
dataset = pd.DataFrame(data, columns = columns)

# convert time into date-time type and other variables into float type
dataset['Time'] = pd.to_datetime(dataset['Time'])
for i in ['Longtitude', 'Latitude', 'Elevation', 'Power', 'Temperature', 'HeartRate', 'Cadence']:
    dataset[i] = dataset[i].astype(float)
dataset.tail()

# duration in seconds
total_time = dataset['Time'][len(dataset) - 1] - dataset['Time'][0]
total_time_in_seconds = total_time.total_seconds()

# total distance in meters using vincenty method
distance_in_meters = []
for i in range(0, len(dataset) - 1):
    starting_point = (dataset['Longtitude'][i], dataset['Latitude'][i])
    ending_point = (dataset['Longtitude'][i+1], dataset['Latitude'][i+1])
    distance = vincenty(starting_point, ending_point).meters
    distance_in_meters.append(distance)
total_distance = sum(distance_in_meters)
print('Total distance of rider 1 is', round(total_distance, 2), 'm')

# Average speed of the rider in m/s
average_speed_ms = round(total_distance / total_time_in_seconds, 2)
print("The average speed of the rider is", average_speed_ms, "m/s")

# Average speed of the rider in m/s
average_speed_kmh = round(average_speed_ms * 18/5, 2)
print("The average speed of the rider is",average_speed_kmh, "km/h")

        count += 1

The number of labs traveled: 18

A scatterplot of GPS coordinates proves that the rider did cycle following a fixed lap. There are several ways to calculate the number of laps completed. The code above is one of the example. However, we are not sure how far the 'distance' variable needs to be less than to be counted as one lap, as in big road, the distance can vary. Therefore, I decided to use visualization to estimate the total distance of each lap.

I come back and tune the threshold. It turns out to be 12.8 meters to return the same number of laps.¶
